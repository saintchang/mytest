```C#
using DynamicExpresso; // 建議透過 NuGet 安裝

public class RuleEvaluator
{
    private readonly Interpreter _interpreter = new Interpreter();

    public string Evaluate(string jsonRule, Dictionary<string, object> context)
    {
        var decisionTable = JsonSerializer.Deserialize<DecisionTable>(jsonRule);
        
        // 將流程變數注入解析器環境
        foreach (var variable in context)
        {
            _interpreter.SetVariable(variable.Key, variable.Value);
        }

        // 遍歷規則，找到第一個符合條件的 (Hit Policy: First)
        foreach (var rule in decisionTable.Rules)
        {
            try 
            {
                // 執行字串運算式判斷
                bool isMatch = _interpreter.Eval<bool>(rule.Condition);
                if (isMatch) return rule.Output;
            }
            catch (Exception ex)
            {
                // 實務建議：記錄 Log 並處理語法錯誤
                Console.WriteLine($"規則解析錯誤: {rule.Condition}, {ex.Message}");
            }
        }

        return "Default"; // 若無匹配則回傳預設值
    }
}
```
